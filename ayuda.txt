

from PyQt5 import QtWidgets, QtGui, uic

app = QtWidgets.QApplication([])

# Cargar el formulario
GUI_MATRIZ = uic.loadUi("ui/Matriz.ui")

# Función para crear y mostrar la matriz
def create_matrix():
    # Obtener el número de filas y columnas del formulario
    rows = int(GUI_MATRIZ.spinBox_rows.value())
    cols = int(GUI_MATRIZ.spinBox_columns.value())

    # Crear la matriz
    matrix = [[0 for _ in range(cols)] for _ in range(rows)]

    # Mostrar la matriz en el formulario
    table_widget = QtWidgets.QTableWidget(rows, cols)
    table_widget.setHorizontalHeaderLabels([str(i) for i in range(cols)])  # Etiquetas de las columnas
    table_widget.setVerticalHeaderLabels([str(i) for i in range(rows)])  # Etiquetas de las filas

    for i in range(rows):
        for j in range(cols):
            item = QtWidgets.QTableWidgetItem(str(matrix[i][j]))
            table_widget.setItem(i, j, item)

    GUI_MATRIZ.layout_matrix.addWidget(table_widget)

# Conectar la función create_matrix al botón del formulario
GUI_MATRIZ.btn_create.clicked.connect(create_matrix)

# Mostrar el formulario
GUI_MATRIZ.show()
app.exec()
